138 CallbackReturn init(
139 const HardwareInfo & hardware_info, rclcpp::Logger logger, rclcpp::Clock::SharedPtr clock)
140 {
141 actuator_clock_ = clock;
142 actuator_logger_ = logger.get_child("hardware_component.actuator." + hardware_info.name);
143 info_ = hardware_info;
144 if (info_.is_async)
145 {
146 RCLCPP_INFO_STREAM(
147 get_logger(), "Starting async handler with scheduler priority: " << info_.thread_priority);
148 async_handler_ = std::make_unique<realtime_tools::AsyncFunctionHandler<return_type>>();
149 async_handler_->init(
150 [this](const rclcpp::Time & time, const rclcpp::Duration & period)
151 {
152 const auto read_start_time = std::chrono::steady_clock::now();
153 const auto ret_read = read(time, period);
154 const auto read_end_time = std::chrono::steady_clock::now();
155 read_return_info_.store(ret_read, std::memory_order_release);
156 read_execution_time_.store(
157 std::chrono::duration_cast<std::chrono::nanoseconds>(read_end_time - read_start_time),
158 std::memory_order_release);
159 if (ret_read != return_type::OK)
160 {
161 return ret_read;
162 }
163 const auto write_start_time = std::chrono::steady_clock::now();
164 const auto ret_write = write(time, period);
165 const auto write_end_time = std::chrono::steady_clock::now();
166 write_return_info_.store(ret_write, std::memory_order_release);
167 write_execution_time_.store(
168 std::chrono::duration_cast<std::chrono::nanoseconds>(write_end_time - write_start_time),
169 std::memory_order_release);
170 return ret_write;
171 },
172 info_.thread_priority);
173 async_handler_->start_thread();
174 }
175 return on_init(hardware_info);
176 };

237 virtual std::vector<StateInterface::ConstSharedPtr> on_export_state_interfaces()
238 {
239 // import the unlisted interfaces
240 std::vector<hardware_interface::InterfaceDescription> unlisted_interface_descriptions =
241 export_unlisted_state_interface_descriptions();
242
243 std::vector<StateInterface::ConstSharedPtr> state_interfaces;
244 state_interfaces.reserve(
245 unlisted_interface_descriptions.size() + joint_state_interfaces_.size());
246
247 // add InterfaceDescriptions and create the StateInterfaces from the descriptions and add to
248 // maps.
249 for (const auto & description : unlisted_interface_descriptions)
250 {
251 auto name = description.get_name();
252 unlisted_state_interfaces_.insert(std::make_pair(name, description));
253 auto state_interface = std::make_shared<StateInterface>(description);
254 actuator_states_.insert(std::make_pair(name, state_interface));
255 unlisted_states_.push_back(state_interface);
256 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
257 }
258
259 for (const auto & [name, descr] : joint_state_interfaces_)
260 {
261 auto state_interface = std::make_shared<StateInterface>(descr);
262 actuator_states_.insert(std::make_pair(name, state_interface));
263 joint_states_.push_back(state_interface);
264 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
265 }
266 return state_interfaces;
267 }

396 HardwareComponentCycleStatus trigger_read(
397 const rclcpp::Time & time, const rclcpp::Duration & period)
398 {
399 HardwareComponentCycleStatus status;
400 status.result = return_type::ERROR;
401 if (info_.is_async)
402 {
403 status.result = read_return_info_.load(std::memory_order_acquire);
404 const auto read_exec_time = read_execution_time_.load(std::memory_order_acquire);
405 if (read_exec_time.count() > 0)
406 {
407 status.execution_time = read_exec_time;
408 }
409 const auto result = async_handler_->trigger_async_callback(time, period);
410 status.successful = result.first;
411 if (!status.successful)
412 {
413 RCLCPP_WARN(
414 get_logger(),
415 "Trigger read/write called while the previous async trigger is still in progress for "
416 "hardware interface : '%s'. Failed to trigger read/write cycle!",
417 info_.name.c_str());
418 status.result = return_type::OK;
419 return status;
420 }
421 }
422 else
423 {
424 const auto start_time = std::chrono::steady_clock::now();
425 status.successful = true;
426 status.result = read(time, period);
427 status.execution_time = std::chrono::duration_cast<std::chrono::nanoseconds>(
428 std::chrono::steady_clock::now() - start_time);
429 }
430 return status;
431 }

455 HardwareComponentCycleStatus trigger_write(
456 const rclcpp::Time & time, const rclcpp::Duration & period)
457 {
458 HardwareComponentCycleStatus status;
459 status.result = return_type::ERROR;
460 if (info_.is_async)
461 {
462 status.successful = true;
463 const auto write_exec_time = write_execution_time_.load(std::memory_order_acquire);
464 if (write_exec_time.count() > 0)
465 {
466 status.execution_time = write_exec_time;
467 }
468 status.result = write_return_info_.load(std::memory_order_acquire);
469 }
470 else
471 {
472 const auto start_time = std::chrono::steady_clock::now();
473 status.successful = true;
474 status.result = write(time, period);
475 status.execution_time = std::chrono::duration_cast<std::chrono::nanoseconds>(
476 std::chrono::steady_clock::now() - start_time);
477 }
478 return status;
