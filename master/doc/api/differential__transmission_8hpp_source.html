171inline DifferentialTransmission::DifferentialTransmission(
172 const std::vector<double> & actuator_reduction, const std::vector<double> & joint_reduction,
173 const std::vector<double> & joint_offset)
174: actuator_reduction_(actuator_reduction),
175 joint_reduction_(joint_reduction),
176 joint_offset_(joint_offset)
177{
178 if (
179 num_actuators() != actuator_reduction_.size() || num_joints() != joint_reduction_.size() ||
180 num_joints() != joint_offset_.size())
181 {
182 throw Exception("Reduction and offset vectors must have size 2.");
183 }
184
185 if (
186 0.0 == actuator_reduction_[0] || 0.0 == actuator_reduction_[1] || 0.0 == joint_reduction_[0] ||
187 0.0 == joint_reduction_[1])
188 {
189 throw Exception("Transmission reduction ratios cannot be zero.");
190 }
191}

193void DifferentialTransmission::configure(
194 const std::vector<JointHandle> & joint_handles,
195 const std::vector<ActuatorHandle> & actuator_handles)
196{
197 if (joint_handles.empty())
198 {
199 throw Exception("No joint handles were passed in");
200 }
201
202 if (actuator_handles.empty())
203 {
204 throw Exception("No actuator handles were passed in");
205 }
206
207 const auto joint_names = get_names(joint_handles);
208 if (joint_names.size() != 2)
209 {
210 throw Exception(
211 "There should be exactly two unique joint names but was given " + to_string(joint_names));
212 }
213 const auto actuator_names = get_names(actuator_handles);
214 if (actuator_names.size() != 2)
215 {
216 throw Exception(
217 "There should be exactly two unique actuator names but was given " +
218 to_string(actuator_names));
219 }
220
221 joint_position_ =
222 get_ordered_handles(joint_handles, joint_names, hardware_interface::HW_IF_POSITION);
223 joint_velocity_ =
224 get_ordered_handles(joint_handles, joint_names, hardware_interface::HW_IF_VELOCITY);
225 joint_effort_ = get_ordered_handles(joint_handles, joint_names, hardware_interface::HW_IF_EFFORT);
226
227 if (joint_position_.size() != 2 && joint_velocity_.size() != 2 && joint_effort_.size() != 2)
228 {
229 throw Exception("Not enough valid or required joint handles were presented.");
230 }
231
232 actuator_position_ =
233 get_ordered_handles(actuator_handles, actuator_names, hardware_interface::HW_IF_POSITION);
234 actuator_velocity_ =
235 get_ordered_handles(actuator_handles, actuator_names, hardware_interface::HW_IF_VELOCITY);
236 actuator_effort_ =
237 get_ordered_handles(actuator_handles, actuator_names, hardware_interface::HW_IF_EFFORT);
238
239 if (
240 actuator_position_.size() != 2 && actuator_velocity_.size() != 2 &&
241 actuator_effort_.size() != 2)
242 {
243 throw Exception(
244 "Not enough valid or required actuator handles were presented. \n" + get_handles_info());
245 }
246
247 if (
248 joint_position_.size() != actuator_position_.size() &&
249 joint_velocity_.size() != actuator_velocity_.size() &&
250 joint_effort_.size() != actuator_effort_.size())
251 {
252 throw Exception("Pair-wise mismatch on interfaces. \n" + get_handles_info());
253 }
254}

256inline void DifferentialTransmission::actuator_to_joint()
257{
258 const auto & ar = actuator_reduction_;
259 const auto & jr = joint_reduction_;
260
261 auto & act_pos = actuator_position_;
262 auto & joint_pos = joint_position_;
263 if (act_pos.size() == num_actuators() && joint_pos.size() == num_joints())
264 {
265 assert(act_pos[0] && act_pos[1] && joint_pos[0] && joint_pos[1]);
266
267 joint_pos[0].set_value(
268 (act_pos[0].get_value() / ar[0] + act_pos[1].get_value() / ar[1]) / (2.0 * jr[0]) +
269 joint_offset_[0]);
270 joint_pos[1].set_value(
271 (act_pos[0].get_value() / ar[0] - act_pos[1].get_value() / ar[1]) / (2.0 * jr[1]) +
272 joint_offset_[1]);
273 }
274
275 auto & act_vel = actuator_velocity_;
276 auto & joint_vel = joint_velocity_;
277 if (act_vel.size() == num_actuators() && joint_vel.size() == num_joints())
278 {
279 assert(act_vel[0] && act_vel[1] && joint_vel[0] && joint_vel[1]);
280
281 joint_vel[0].set_value(
282 (act_vel[0].get_value() / ar[0] + act_vel[1].get_value() / ar[1]) / (2.0 * jr[0]));
283 joint_vel[1].set_value(
284 (act_vel[0].get_value() / ar[0] - act_vel[1].get_value() / ar[1]) / (2.0 * jr[1]));
285 }
286
287 auto & act_eff = actuator_effort_;
288 auto & joint_eff = joint_effort_;
289 if (act_eff.size() == num_actuators() && joint_eff.size() == num_joints())
290 {
291 assert(act_eff[0] && act_eff[1] && joint_eff[0] && joint_eff[1]);
292
293 joint_eff[0].set_value(
294 jr[0] * (act_eff[0].get_value() * ar[0] + act_eff[1].get_value() * ar[1]));
295 joint_eff[1].set_value(
296 jr[1] * (act_eff[0].get_value() * ar[0] - act_eff[1].get_value() * ar[1]));
297 }
298}

300inline void DifferentialTransmission::joint_to_actuator()
301{
302 const auto & ar = actuator_reduction_;
303 const auto & jr = joint_reduction_;
304
305 auto & act_pos = actuator_position_;
306 auto & joint_pos = joint_position_;
307 if (act_pos.size() == num_actuators() && joint_pos.size() == num_joints())
308 {
309 assert(act_pos[0] && act_pos[1] && joint_pos[0] && joint_pos[1]);
310
311 double joints_offset_applied[2] = {
312 joint_pos[0].get_value() - joint_offset_[0], joint_pos[1].get_value() - joint_offset_[1]};
313 act_pos[0].set_value(
314 (joints_offset_applied[0] * jr[0] + joints_offset_applied[1] * jr[1]) * ar[0]);
315 act_pos[1].set_value(
316 (joints_offset_applied[0] * jr[0] - joints_offset_applied[1] * jr[1]) * ar[1]);
317 }
318
319 auto & act_vel = actuator_velocity_;
320 auto & joint_vel = joint_velocity_;
321 if (act_vel.size() == num_actuators() && joint_vel.size() == num_joints())
322 {
323 assert(act_vel[0] && act_vel[1] && joint_vel[0] && joint_vel[1]);
324
325 act_vel[0].set_value(
326 (joint_vel[0].get_value() * jr[0] + joint_vel[1].get_value() * jr[1]) * ar[0]);
327 act_vel[1].set_value(
328 (joint_vel[0].get_value() * jr[0] - joint_vel[1].get_value() * jr[1]) * ar[1]);
329 }
330
331 auto & act_eff = actuator_effort_;
332 auto & joint_eff = joint_effort_;
333 if (act_eff.size() == num_actuators() && joint_eff.size() == num_joints())
334 {
335 assert(act_eff[0] && act_eff[1] && joint_eff[0] && joint_eff[1]);
336
337 act_eff[0].set_value(
338 (joint_eff[0].get_value() / jr[0] + joint_eff[1].get_value() / jr[1]) / (2.0 * ar[0]));
339 act_eff[1].set_value(
340 (joint_eff[0].get_value() / jr[0] - joint_eff[1].get_value() / jr[1]) / (2.0 * ar[1]));
341 }
342}
