110 CallbackReturn init(
111 const HardwareInfo & hardware_info, rclcpp::Logger logger,
112 rclcpp::node_interfaces::NodeClockInterface::SharedPtr clock_interface)
113 {
114 clock_interface_ = clock_interface;
115 actuator_logger_ = logger.get_child("hardware_component.actuator." + hardware_info.name);
116 info_ = hardware_info;
117 if (info_.is_async)
118 {
119 RCLCPP_INFO_STREAM(
120 get_logger(), "Starting async handler with scheduler priority: " << info_.thread_priority);
121 async_handler_ = std::make_unique<realtime_tools::AsyncFunctionHandler<return_type>>();
122 async_handler_->init(
123 [this](const rclcpp::Time & time, const rclcpp::Duration & period)
124 {
125 if (next_trigger_ == TriggerType::READ)
126 {
127 const auto ret = read(time, period);
128 next_trigger_ = TriggerType::WRITE;
129 return ret;
130 }
131 else
132 {
133 const auto ret = write(time, period);
134 next_trigger_ = TriggerType::READ;
135 return ret;
136 }
137 },
138 info_.thread_priority);
139 async_handler_->start_thread();
140 }
141 return on_init(hardware_info);
142 };

203 virtual std::vector<StateInterface::ConstSharedPtr> on_export_state_interfaces()
204 {
205 // import the unlisted interfaces
206 std::vector<hardware_interface::InterfaceDescription> unlisted_interface_descriptions =
207 export_unlisted_state_interface_descriptions();
208
209 std::vector<StateInterface::ConstSharedPtr> state_interfaces;
210 state_interfaces.reserve(
211 unlisted_interface_descriptions.size() + joint_state_interfaces_.size());
212
213 // add InterfaceDescriptions and create the StateInterfaces from the descriptions and add to
214 // maps.
215 for (const auto & description : unlisted_interface_descriptions)
216 {
217 auto name = description.get_name();
218 unlisted_state_interfaces_.insert(std::make_pair(name, description));
219 auto state_interface = std::make_shared<StateInterface>(description);
220 actuator_states_.insert(std::make_pair(name, state_interface));
221 unlisted_states_.push_back(state_interface);
222 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
223 }
224
225 for (const auto & [name, descr] : joint_state_interfaces_)
226 {
227 auto state_interface = std::make_shared<StateInterface>(descr);
228 actuator_states_.insert(std::make_pair(name, state_interface));
229 joint_states_.push_back(state_interface);
230 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
231 }
232 return state_interfaces;
233 }

362 return_type trigger_read(const rclcpp::Time & time, const rclcpp::Duration & period)
363 {
364 return_type result = return_type::ERROR;
365 if (info_.is_async)
366 {
367 bool trigger_status = true;
368 if (next_trigger_ == TriggerType::WRITE)
369 {
370 RCLCPP_WARN(
371 get_logger(),
372 "Trigger read called while write async handler call is still pending for hardware "
373 "interface : '%s'. Skipping read cycle and will wait for a write cycle!",
374 info_.name.c_str());
375 return return_type::OK;
376 }
377 std::tie(trigger_status, result) = async_handler_->trigger_async_callback(time, period);
378 if (!trigger_status)
379 {
380 RCLCPP_WARN(
381 get_logger(),
382 "Trigger read called while write async trigger is still in progress for hardware "
383 "interface : '%s'. Failed to trigger read cycle!",
384 info_.name.c_str());
385 return return_type::OK;
386 }
387 }
388 else
389 {
390 result = read(time, period);
391 }
392 return result;
393 }
