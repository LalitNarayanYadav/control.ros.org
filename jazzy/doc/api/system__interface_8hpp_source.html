142 CallbackReturn init(
143 const HardwareInfo & hardware_info, rclcpp::Logger logger, rclcpp::Clock::SharedPtr clock)
144 {
145 system_clock_ = clock;
146 system_logger_ = logger.get_child("hardware_component.system." + hardware_info.name);
147 info_ = hardware_info;
148 if (info_.is_async)
149 {
150 RCLCPP_INFO_STREAM(
151 get_logger(), "Starting async handler with scheduler priority: " << info_.thread_priority);
152 async_handler_ = std::make_unique<realtime_tools::AsyncFunctionHandler<return_type>>();
153 async_handler_->init(
154 [this](const rclcpp::Time & time, const rclcpp::Duration & period)
155 {
156 const auto read_start_time = std::chrono::steady_clock::now();
157 const auto ret_read = read(time, period);
158 const auto read_end_time = std::chrono::steady_clock::now();
159 read_return_info_.store(ret_read, std::memory_order_release);
160 read_execution_time_.store(
161 std::chrono::duration_cast<std::chrono::nanoseconds>(read_end_time - read_start_time),
162 std::memory_order_release);
163 if (ret_read != return_type::OK)
164 {
165 return ret_read;
166 }
167 const auto write_start_time = std::chrono::steady_clock::now();
168 const auto ret_write = write(time, period);
169 const auto write_end_time = std::chrono::steady_clock::now();
170 write_return_info_.store(ret_write, std::memory_order_release);
171 write_execution_time_.store(
172 std::chrono::duration_cast<std::chrono::nanoseconds>(write_end_time - write_start_time),
173 std::memory_order_release);
174 return ret_write;
175 },
176 info_.thread_priority);
177 async_handler_->start_thread();
178 }
179 return on_init(hardware_info);
180 };

243 virtual std::vector<StateInterface::ConstSharedPtr> on_export_state_interfaces()
244 {
245 // import the unlisted interfaces
246 std::vector<hardware_interface::InterfaceDescription> unlisted_interface_descriptions =
247 export_unlisted_state_interface_descriptions();
248
249 std::vector<StateInterface::ConstSharedPtr> state_interfaces;
250 state_interfaces.reserve(
251 unlisted_interface_descriptions.size() + joint_state_interfaces_.size() +
252 sensor_state_interfaces_.size() + gpio_state_interfaces_.size());
253
254 // add InterfaceDescriptions and create the StateInterfaces from the descriptions and add to
255 // maps.
256 for (const auto & description : unlisted_interface_descriptions)
257 {
258 auto name = description.get_name();
259 unlisted_state_interfaces_.insert(std::make_pair(name, description));
260 auto state_interface = std::make_shared<StateInterface>(description);
261 system_states_.insert(std::make_pair(name, state_interface));
262 unlisted_states_.push_back(state_interface);
263 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
264 }
265
266 for (const auto & [name, descr] : joint_state_interfaces_)
267 {
268 auto state_interface = std::make_shared<StateInterface>(descr);
269 system_states_.insert(std::make_pair(name, state_interface));
270 joint_states_.push_back(state_interface);
271 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
272 }
273 for (const auto & [name, descr] : sensor_state_interfaces_)

426 HardwareComponentCycleStatus trigger_read(
427 const rclcpp::Time & time, const rclcpp::Duration & period)
428 {
429 HardwareComponentCycleStatus status;
430 status.result = return_type::ERROR;
431 if (info_.is_async)
432 {
433 status.result = read_return_info_.load(std::memory_order_acquire);
434 const auto read_exec_time = read_execution_time_.load(std::memory_order_acquire);
435 if (read_exec_time.count() > 0)
436 {
437 status.execution_time = read_exec_time;
438 }
439 const auto result = async_handler_->trigger_async_callback(time, period);
440 status.successful = result.first;
441 if (!status.successful)
442 {
443 RCLCPP_WARN(
444 get_logger(),
445 "Trigger read/write called while the previous async trigger is still in progress for "
446 "hardware interface : '%s'. Failed to trigger read/write cycle!",
447 info_.name.c_str());
448 status.result = return_type::OK;
449 return status;
450 }
451 }
452 else
453 {
454 const auto start_time = std::chrono::steady_clock::now();
455 status.successful = true;
456 status.result = read(time, period);
457 status.execution_time = std::chrono::duration_cast<std::chrono::nanoseconds>(
458 std::chrono::steady_clock::now() - start_time);
459 }
460 return status;
461 }

485 HardwareComponentCycleStatus trigger_write(
486 const rclcpp::Time & time, const rclcpp::Duration & period)
487 {
488 HardwareComponentCycleStatus status;
489 status.result = return_type::ERROR;
490 if (info_.is_async)
491 {
492 status.successful = true;
493 const auto write_exec_time = write_execution_time_.load(std::memory_order_acquire);
494 if (write_exec_time.count() > 0)
495 {
496 status.execution_time = write_exec_time;
497 }
498 status.result = write_return_info_.load(std::memory_order_acquire);
499 }
500 else
501 {
502 const auto start_time = std::chrono::steady_clock::now();
503 status.successful = true;
504 status.result = write(time, period);
505 status.execution_time = std::chrono::duration_cast<std::chrono::nanoseconds>(
506 std::chrono::steady_clock::now() - start_time);
507 }
508 return status;
509 }
