122 CallbackReturn init(
123 const HardwareInfo & hardware_info, rclcpp::Logger logger, rclcpp::Clock::SharedPtr clock)
124 {
125 sensor_clock_ = clock;
126 sensor_logger_ = logger.get_child("hardware_component.sensor." + hardware_info.name);
127 info_ = hardware_info;
128 if (info_.is_async)
129 {
130 RCLCPP_INFO_STREAM(
131 get_logger(), "Starting async handler with scheduler priority: " << info_.thread_priority);
132 read_async_handler_ = std::make_unique<realtime_tools::AsyncFunctionHandler<return_type>>();
133 read_async_handler_->init(
134 std::bind(&SensorInterface::read, this, std::placeholders::_1, std::placeholders::_2),
135 info_.thread_priority);
136 read_async_handler_->start_thread();
137 }
138 return on_init(hardware_info);
139 };

199 virtual std::vector<StateInterface::ConstSharedPtr> on_export_state_interfaces()
200 {
201 // import the unlisted interfaces
202 std::vector<hardware_interface::InterfaceDescription> unlisted_interface_descriptions =
203 export_unlisted_state_interface_descriptions();
204
205 std::vector<StateInterface::ConstSharedPtr> state_interfaces;
206 state_interfaces.reserve(
207 unlisted_interface_descriptions.size() + sensor_state_interfaces_.size() +
208 joint_state_interfaces_.size());
209
210 // add InterfaceDescriptions and create StateInterfaces from the descriptions and add to maps.
211 for (const auto & description : unlisted_interface_descriptions)
212 {
213 auto name = description.get_name();
214 unlisted_state_interfaces_.insert(std::make_pair(name, description));
215 auto state_interface = std::make_shared<StateInterface>(description);
216 sensor_states_map_.insert(std::make_pair(name, state_interface));
217 unlisted_states_.push_back(state_interface);
218 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
219 }
220
221 for (const auto & [name, descr] : sensor_state_interfaces_)
222 {
223 // TODO(Manuel) check for duplicates otherwise only the first appearance of "name" is inserted
224 auto state_interface = std::make_shared<StateInterface>(descr);
225 sensor_states_map_.insert(std::make_pair(name, state_interface));
226 sensor_states_.push_back(state_interface);
227 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
228 }
229
230 for (const auto & [name, descr] : joint_state_interfaces_)
231 {
232 auto state_interface = std::make_shared<StateInterface>(descr);
233 sensor_states_map_.insert(std::make_pair(name, state_interface));
234 joint_states_.push_back(state_interface);
235 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
236 }
237
238 return state_interfaces;
239 }

251 HardwareComponentCycleStatus trigger_read(
252 const rclcpp::Time & time, const rclcpp::Duration & period)
253 {
254 HardwareComponentCycleStatus status;
255 status.result = return_type::ERROR;
256 if (info_.is_async)
257 {
258 const auto result = read_async_handler_->trigger_async_callback(time, period);
259 status.successful = result.first;
260 status.result = result.second;
261 const auto execution_time = read_async_handler_->get_last_execution_time();
262 if (execution_time.count() > 0)
263 {
264 status.execution_time = execution_time;
265 }
266 if (!status.successful)
267 {
268 RCLCPP_WARN(
269 get_logger(),
270 "Trigger read called while read async handler is still in progress for hardware "
271 "interface : '%s'. Failed to trigger read cycle!",
272 info_.name.c_str());
273 status.result = return_type::OK;
274 return status;
275 }
276 }
277 else
278 {
279 const auto start_time = std::chrono::steady_clock::now();
280 status.successful = true;
281 status.result = read(time, period);
282 status.execution_time = std::chrono::duration_cast<std::chrono::nanoseconds>(
283 std::chrono::steady_clock::now() - start_time);
284 }
285 return status;
286 }
