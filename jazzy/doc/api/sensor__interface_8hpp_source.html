109 CallbackReturn init(
110 const HardwareInfo & hardware_info, rclcpp::Logger logger,
111 rclcpp::node_interfaces::NodeClockInterface::SharedPtr clock_interface)
112 {
113 clock_interface_ = clock_interface;
114 sensor_logger_ = logger.get_child("hardware_component.sensor." + hardware_info.name);
115 info_ = hardware_info;
116 if (info_.is_async)
117 {
118 RCLCPP_INFO_STREAM(
119 get_logger(), "Starting async handler with scheduler priority: " << info_.thread_priority);
120 read_async_handler_ = std::make_unique<realtime_tools::AsyncFunctionHandler<return_type>>();
121 read_async_handler_->init(
122 std::bind(&SensorInterface::read, this, std::placeholders::_1, std::placeholders::_2),
123 info_.thread_priority);
124 read_async_handler_->start_thread();
125 }
126 return on_init(hardware_info);
127 };

187 virtual std::vector<StateInterface::ConstSharedPtr> on_export_state_interfaces()
188 {
189 // import the unlisted interfaces
190 std::vector<hardware_interface::InterfaceDescription> unlisted_interface_descriptions =
191 export_unlisted_state_interface_descriptions();
192
193 std::vector<StateInterface::ConstSharedPtr> state_interfaces;
194 state_interfaces.reserve(
195 unlisted_interface_descriptions.size() + sensor_state_interfaces_.size() +
196 joint_state_interfaces_.size());
197
198 // add InterfaceDescriptions and create StateInterfaces from the descriptions and add to maps.
199 for (const auto & description : unlisted_interface_descriptions)
200 {
201 auto name = description.get_name();
202 unlisted_state_interfaces_.insert(std::make_pair(name, description));
203 auto state_interface = std::make_shared<StateInterface>(description);
204 sensor_states_map_.insert(std::make_pair(name, state_interface));
205 unlisted_states_.push_back(state_interface);
206 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
207 }
208
209 for (const auto & [name, descr] : sensor_state_interfaces_)
210 {
211 // TODO(Manuel) check for duplicates otherwise only the first appearance of "name" is inserted
212 auto state_interface = std::make_shared<StateInterface>(descr);
213 sensor_states_map_.insert(std::make_pair(name, state_interface));
214 sensor_states_.push_back(state_interface);
215 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
216 }
217
218 for (const auto & [name, descr] : joint_state_interfaces_)
219 {
220 auto state_interface = std::make_shared<StateInterface>(descr);
221 sensor_states_map_.insert(std::make_pair(name, state_interface));
222 joint_states_.push_back(state_interface);
223 state_interfaces.push_back(std::const_pointer_cast<const StateInterface>(state_interface));
224 }
225
226 return state_interfaces;
227 }

239 return_type trigger_read(const rclcpp::Time & time, const rclcpp::Duration & period)
240 {
241 return_type result = return_type::ERROR;
242 if (info_.is_async)
243 {
244 bool trigger_status = true;
245 std::tie(trigger_status, result) = read_async_handler_->trigger_async_callback(time, period);
246 if (!trigger_status)
247 {
248 RCLCPP_WARN(
249 get_logger(),
250 "Trigger read called while read async handler is still in progress for hardware "
251 "interface : '%s'. Failed to trigger read cycle!",
252 info_.name.c_str());
253 return return_type::OK;
254 }
255 }
256 else
257 {
258 result = read(time, period);
259 }
260 return result;
261 }
