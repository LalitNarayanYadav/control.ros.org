136class HandleDataType
137{
138public:
139 enum Value : int8_t
140 {
141 UNKNOWN = -1,
142 DOUBLE,
143 BOOL
144 };
145
146 HandleDataType() = default;
147 constexpr HandleDataType(Value value) : value_(value) {} // NOLINT(runtime/explicit)
148 explicit HandleDataType(const std::string & data_type)
149 {
150 if (data_type == "double")
151 {
152 value_ = DOUBLE;
153 }
154 else if (data_type == "bool")
155 {
156 value_ = BOOL;
157 }
158 else
159 {
160 value_ = UNKNOWN;
161 }
162 }
163
164 operator Value() const { return value_; }
165
166 explicit operator bool() const = delete;
167
168 constexpr bool operator==(HandleDataType other) const { return value_ == other.value_; }
169 constexpr bool operator!=(HandleDataType other) const { return value_ != other.value_; }
170
171 constexpr bool operator==(Value other) const { return value_ == other; }
172 constexpr bool operator!=(Value other) const { return value_ != other; }
173
174 std::string to_string() const
175 {
176 switch (value_)
177 {
178 case DOUBLE:
179 return "double";
180 case BOOL:
181 return "bool";
182 default:
183 return "unknown";
184 }
185 }
186
187 HandleDataType from_string(const std::string & data_type) { return HandleDataType(data_type); }
188
189private:
190 Value value_ = UNKNOWN;
191};
